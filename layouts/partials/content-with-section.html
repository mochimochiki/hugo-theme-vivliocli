(printf "${1}%s ${2}" $h2section) }} {{- $h2replaced = partial "functions/addPdftoc" $h2replaced }} {{- $content = (replace $content . $h2replaced) | safeHTML }} {{- if lt (add $pageLevel 1) $settingLevel }} {{/* h2 - Next h2/EOF までの間にある h3 について章番号付加 */}} {{- $h2content := $content | findRE (printf "%s(.|\n)*?(|\\z)" $h2replaced) }} {{- $h3List := findRE "(.|\n)*?" (printf "${1}%s ${2}" $h3section) }} {{- $h3replaced = partial "functions/addPdftoc" $h3replaced }} {{- $content = $content | replaceRE . $h3replaced | safeHTML }} {{- end }} {{/* {{- partial "log" (printf "%s" $h2content)}} */}} {{- end }} {{- end }} {{- end }} {{ $content }}
